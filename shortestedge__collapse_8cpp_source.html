27void shortestedge_collapse(TriMesh& mesh, const ShortestEdgeCollapseOptions& options)
28{
29 if (mesh.top_simplex_type() != PrimitiveType::Triangle) {
30 log_and_throw_error(
31 "shortest edge collapse works only for triangle meshes: {}",
32 primitive_type_name(mesh.top_simplex_type()));
33 }
34
35 const attribute::MeshAttributeHandle& position_handle = options.position_handle;
36 const attribute::MeshAttributeHandle& inversion_position_handle =
37 options.inversion_position_handle;
38
39 // TriMesh& mesh = static_cast<TriMesh&>(position_handle.mesh());
40
41 std::vector<attribute::MeshAttributeHandle> pass_through_attributes =
42 options.pass_through_attributes;
43
45
46 auto visited_edge_flag =
47 mesh.register_attribute<char>("visited_edge", PrimitiveType::Edge, 1, false, char(1));
48
49 auto update_flag_func = [](Eigen::Ref<const Eigen::MatrixXd> P) -> Eigen::VectorX<char> {
50 assert(P.cols() == 2);
51 assert(P.rows() == 2 || P.rows() == 3);
52 return Eigen::VectorX<char>::Constant(1, char(1));
53 };
54 auto tag_update =
55 std::make_shared<wmtk::operations::SingleAttributeTransferStrategy<char, double>>(
56 visited_edge_flag,
57 position_handle,
58 update_flag_func);
59
61 // Storing edge lengths
62 auto edge_length_attribute =
63 mesh.register_attribute<double>("edge_length", PrimitiveType::Edge, 1);
64 auto edge_length_accessor = mesh.create_accessor(edge_length_attribute.as<double>());
65 // Edge length update
66 auto compute_edge_length = [](Eigen::Ref<const Eigen::MatrixXd> P) -> Eigen::VectorXd {
67 assert(P.cols() == 2);
68 assert(P.rows() == 2 || P.rows() == 3);
69 return Eigen::VectorXd::Constant(1, (P.col(0) - P.col(1)).norm());
70 };
71 auto edge_length_update =
72 std::make_shared<wmtk::operations::SingleAttributeTransferStrategy<double, double>>(
73 edge_length_attribute,
74 position_handle,
75 compute_edge_length);
76 edge_length_update->run_on_all();
77
79 // computing bbox diagonal
80 Eigen::VectorXd bmin(position_handle.dimension());
81 bmin.setConstant(std::numeric_limits<double>::max());
82 Eigen::VectorXd bmax(position_handle.dimension());
83 bmax.setConstant(std::numeric_limits<double>::lowest());
84
85 auto pt_accessor = mesh.create_const_accessor<double>(position_handle);
86
87 const auto vertices = mesh.get_all(PrimitiveType::Vertex);
88 for (const auto& v : vertices) {
89 const auto p = pt_accessor.vector_attribute(v);
90 for (int64_t d = 0; d < bmax.size(); ++d) {
91 bmin[d] = std::min(bmin[d], p[d]);
92 bmax[d] = std::max(bmax[d], p[d]);
93 }
94 }
95
96 const double bbdiag = (bmax - bmin).norm();
97
98 const double length_abs = bbdiag * options.length_rel;
99
100 wmtk::logger().info(
101 "bbox max {}, bbox min {}, diag {}, target edge length {}",
102 bmax,
103 bmin,
104 bbdiag,
105 length_abs);
106
107 auto short_edges_first_priority = [&](const simplex::Simplex& s) {
108 assert(s.primitive_type() == PrimitiveType::Edge);
109 return edge_length_accessor.const_scalar_attribute(s.tuple());
110 };
111 pass_through_attributes.push_back(edge_length_attribute);
112 auto todo = std::make_shared<TodoSmallerInvariant>(
113 mesh,
114 edge_length_attribute.as<double>(),
115 4. / 5. * length_abs); // MTAO: why is this 4/5?
116
118
119 auto invariant_link_condition = std::make_shared<MultiMeshLinkConditionInvariant>(mesh);
120
121 auto invariant_interior_edge = std::make_shared<invariants::InvariantCollection>(mesh);
122 auto invariant_interior_vertex = std::make_shared<invariants::InvariantCollection>(mesh);
123
124 auto set_all_invariants = [&](auto&& m) {
125 invariant_interior_edge->add(
126 std::make_shared<invariants::InteriorSimplexInvariant>(m, PrimitiveType::Edge));
127 invariant_interior_vertex->add(
128 std::make_shared<invariants::InteriorSimplexInvariant>(m, PrimitiveType::Vertex));
129 };
130 multimesh::MultiMeshVisitor visitor(set_all_invariants);
131 visitor.execute_from_root(mesh);
132
133 auto invariant_mm_map = std::make_shared<MultiMeshMapValidInvariant>(mesh);
134
136 std::vector<attribute::MeshAttributeHandle> position_handles;
137 position_handles.push_back(position_handle);
138 if (inversion_position_handle.is_valid()) {
139 position_handles.push_back(inversion_position_handle);
140 }
141
143 // collapse
144 auto collapse = std::make_shared<wmtk::operations::EdgeCollapse>(mesh);
145 collapse->add_invariant(todo);
146 collapse->add_invariant(invariant_link_condition);
147 collapse->add_invariant(invariant_mm_map);
148
149
150 if (options.envelope_size) {
151 collapse->add_invariant(std::make_shared<wmtk::invariants::EnvelopeInvariant>(
152 position_handle,
153 bbdiag * options.envelope_size.value(),
154 position_handle));
155 }
156
157
158 if (inversion_position_handle.is_valid()) {
159 collapse->add_invariant(std::make_shared<SimplexInversionInvariant<double>>(
160 inversion_position_handle.mesh(),
161 inversion_position_handle.as<double>()));
162 }
163 collapse->set_new_attribute_strategy(
164 visited_edge_flag,
165 wmtk::operations::CollapseBasicStrategy::None);
166 collapse->add_transfer_strategy(tag_update);
167
168 collapse->set_priority(short_edges_first_priority);
169 collapse->add_transfer_strategy(edge_length_update);
170
171 if (options.lock_boundary) {
172 collapse->add_invariant(invariant_interior_edge);
173 // set collapse towards boundary
174 for (const auto& pos_handle : position_handles) {
175 auto pos_collapse_strategy =
176 std::make_shared<wmtk::operations::CollapseNewAttributeStrategy<double>>(
177 pos_handle);
178 pos_collapse_strategy->set_strategy(wmtk::operations::CollapseBasicStrategy::CopyOther);
179 pos_collapse_strategy->set_simplex_predicate(
180 wmtk::operations::BasicSimplexPredicate::IsInterior);
181 collapse->set_new_attribute_strategy(pos_handle, pos_collapse_strategy);
182 }
183 } else {
184 for (const auto& pos_handle : position_handles) {
185 collapse->set_new_attribute_strategy(
186 pos_handle,
187 wmtk::operations::CollapseBasicStrategy::CopyOther);
188 }
189 }
190
191
192 for (const auto& attr : pass_through_attributes) {
193 collapse->set_new_attribute_strategy(attr);
194 }
195
196
198 Scheduler scheduler;
199 SchedulerStats pass_stats =
200 scheduler.run_operation_on_all(*collapse, visited_edge_flag.as<char>());
201
202
203 // // debug code
204 // for (const auto& e : mesh.get_all(PrimitiveType::Edge)) {
205 // if (mesh.is_boundary(PrimitiveType::Edge, e)) {
206 // logger().error("after sec mesh has nonmanifold edges");
207 // }
208 // }
209
210 multimesh::consolidate(mesh);
211
212 logger().info(
213 "Executed {} ops (S/F) {}/{}. Time: collecting: {}, sorting: {}, executing: {}",
214 pass_stats.number_of_performed_operations(),
215 pass_stats.number_of_successful_operations(),
216 pass_stats.number_of_failed_operations(),
217 pass_stats.collecting_time,
218 pass_stats.sorting_time,
219 pass_stats.executing_time);
220}
