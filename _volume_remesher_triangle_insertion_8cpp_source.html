33void generate_background_mesh(
34 const Vector3d& bbox_min,
35 const Vector3d& bbox_max,
36 const std::array<int64_t, 3>& resolution, // number of cells in each dimension
37 RowVectors4l& background_TV,
38 RowVectors3d& background_V)

40 // resize matrices
41 background_TV.resize(6 * resolution[0] * resolution[1] * resolution[2], 4);

44 // fill in tv
45 int64_t tet_index = 0;
46

48 for (int64_t j = 0; j < resolution[1]; ++j) {
49 for (int64_t i = 0; i < resolution[0]; ++i) {
50 int64_t cell_v[8] = {
51 v_index(resolution, {{i, j, k}}),

55 v_index(resolution, {{i, j, k + 1}}),

58 v_index(resolution, {{i, j + 1, k + 1}})};
59 background_TV.row(tet_index++) << cell_v[0], cell_v[1], cell_v[3], cell_v[4];
60 background_TV.row(tet_index++) << cell_v[5], cell_v[2], cell_v[6], cell_v[7];
61 background_TV.row(tet_index++) << cell_v[4], cell_v[1], cell_v[5], cell_v[3];
62 background_TV.row(tet_index++) << cell_v[4], cell_v[3], cell_v[7], cell_v[5];
63 background_TV.row(tet_index++) << cell_v[3], cell_v[1], cell_v[5], cell_v[2];
64 background_TV.row(tet_index++) << cell_v[2], cell_v[3], cell_v[7], cell_v[5];

73 for (int64_t j = 0; j < resolution[1] + 1; ++j) {
74 for (int64_t i = 0; i < resolution[0] + 1; ++i) {

78 bbox_min[2] + (bbox_max[2] - bbox_min[2]) * k / resolution[2];

84std::vector<std::array<int64_t, 3>> triangulate_polygon_face(std::vector<Vector3r> points)
85{
86 // triangulate weak convex polygons
87 std::vector<std::array<int64_t, 3>> triangulated_faces;
88
89 std::vector<std::pair<Vector3r, int64_t>> points_vector;
90 for (int64_t i = 0; i < points.size(); ++i) {
91 points_vector.push_back(std::pair<Vector3r, int64_t>(points[i], i));
92 }
93
94 // find the first colinear ABC with nonlinear BCD and delete C from vector
95 while (points_vector.size() > 3) {
96 bool no_colinear = true;
97 for (int64_t i = 0; i < points_vector.size(); ++i) {
98 auto cur = points_vector[i];
99 auto next = points_vector[(i + 1) % points_vector.size()];
100 auto prev = points_vector[(i + points_vector.size() - 1) % points_vector.size()];
101 auto nextnext = points_vector[(i + 2) % points_vector.size()];
102
103 // TODO: check if orient2d == 0 works for check colinearity
104 // if (orient2d(prev.first.data(), cur.first.data(), next.first.data()) == 0 &&
105 // orient2d(cur.first.data(), next.first.data(), nextnext.first.data()) != 0) {
106
107 Vector3r a = cur.first - prev.first;
108 Vector3r b = next.first - cur.first;
109 Vector3r c = nextnext.first - next.first;
110
111 if (((a[0] * b[1] - a[1] * b[0]).get_sign() == 0 &&
112 (a[1] * b[2] - a[2] * b[1]).get_sign() == 0 &&
113 (a[0] * b[2] - a[2] * b[0]).get_sign() == 0) &&
114 (((b[0] * c[1] - b[1] * c[0]).get_sign() != 0 ||
115 (b[1] * c[2] - b[2] * c[1]).get_sign() != 0 ||
116 (b[0] * c[2] - b[2] * c[0]).get_sign() != 0))) {
117 no_colinear = false;
118 std::array<int64_t, 3> t = {{cur.second, next.second, nextnext.second}};
119 triangulated_faces.push_back(t);
120 points_vector.erase(points_vector.begin() + ((i + 1) % points_vector.size()));
121 break;
122 } else {
123 continue;
124 }
125 }
126
127 if (no_colinear) break;
128 }
129
130 // cleanup convex polygon
131 while (points_vector.size() >= 3) {
132 std::array<int64_t, 3> t = {
133 {points_vector[0].second,
134 points_vector[1].second,
135 points_vector[points_vector.size() - 1].second}};
136 triangulated_faces.push_back(t);
137 points_vector.erase(points_vector.begin());
138 }
139
140 return triangulated_faces;
141}

152 Vector3d bbox_min = V.colwise().minCoeff();
153 Vector3d bbox_max = V.colwise().maxCoeff();
154 double diag_length = (bbox_max - bbox_min).norm();
155 double eps = 1 / 15.0; // TODO: set this flexible
156 double delta = diag_length * eps;
157 bbox_min -= Vector3d(delta, delta, delta);
158 bbox_max += Vector3d(delta, delta, delta);
159 double target_edge_length = diag_length * eps_target;

161 // compute resolution
162 std::array<int64_t, 3> resolution = {
163 {(int64_t)((bbox_max[0] - bbox_min[0]) / target_edge_length),

167 // generate background mesh
168 RowVectors4l background_TV;

173 wmtk::logger().info(
174 "generated background mesh with resolution {} {} {}",
175 resolution[0],
176 resolution[1],
177 resolution[2]);
178 } else {
179 assert(bgT.size() != 0);
180 background_TV = bgT;
181 background_V = bgV;

186 // inputs
187 std::vector<double> tri_vrt_coords(V.rows() * 3);
188 std::vector<uint32_t> triangle_indices(F.rows() * 3);
189 std::vector<double> tet_vrt_coords(background_V.rows() * 3);
190 std::vector<uint32_t> tet_indices(background_TV.rows() * 4);
191
192 for (int64_t i = 0; i < V.rows(); ++i) {
193 tri_vrt_coords[i * 3 + 0] = V(i, 0);
194 tri_vrt_coords[i * 3 + 1] = V(i, 1);
195 tri_vrt_coords[i * 3 + 2] = V(i, 2);
196 }
197
198 for (int64_t i = 0; i < F.rows(); ++i) {
199 triangle_indices[i * 3 + 0] = F(i, 0);
200 triangle_indices[i * 3 + 1] = F(i, 1);
201 triangle_indices[i * 3 + 2] = F(i, 2);

204 for (int64_t i = 0; i < background_V.rows(); ++i) {
205 tet_vrt_coords[i * 3 + 0] = background_V(i, 0);
206 tet_vrt_coords[i * 3 + 1] = background_V(i, 1);
207 tet_vrt_coords[i * 3 + 2] = background_V(i, 2);

211 tet_indices[i * 4 + 0] = background_TV(i, 0);
212 tet_indices[i * 4 + 1] = background_TV(i, 1);

260 for (int64_t i = 0; i < embedded_facets.size(); ++i) {
261 int64_t polysize = embedded_facets[i];

264 polygon.push_back(embedded_facets[j]);

266 polygon_faces.push_back(polygon);
267 i += polysize; // dangerous behavior, but working now

273 for (int64_t j = i + 1; j <= i + cellsize; ++j) {

276 polygon_cells.push_back(polygon_cell);
277 i += cellsize; // dangerous behavior, but working now

280 polygon_faces_on_input.resize(polygon_faces.size(), false);

282 for (int64_t i = 0; i < embedded_facets_on_input.size(); ++i) {
283 polygon_faces_on_input[embedded_facets_on_input[i]] = true;
284 }

286 // triangulate polygon faces
287 std::vector<std::array<int64_t, 3>> triangulated_faces;
288 std::vector<bool> triangulated_faces_on_input;
289 std::vector<std::vector<int64_t>> map_poly_to_tri_face(polygon_faces.size());

291 for (int64_t i = 0; i < polygon_faces.size(); ++i) {
292 // already clipped in other polygon

371 int64_t v1 = polygon_faces[polygon_cell[0]][1];
372 int64_t v2 = polygon_faces[polygon_cell[0]][2];

519 v_coords_final_rational[t[1]],

530

532 V_final.row(i) = v_coords_final[i];
533 }
534
535 for (int64_t i = 0; i < v_coords_final_rational.size(); ++i) {
536 V_final_rational.row(i) = v_coords_final_rational[i];
537 }
538
539 for (int64_t i = 0; i < tets_final.size(); ++i) {
540 TV_final.row(i) << tets_final[i][0], tets_final[i][1], tets_final[i][2], tets_final[i][3];

551 wmtk::logger().info("init tetmesh finished.");
552
553 if (!m->is_connectivity_valid()) {
554 wmtk::logger().error("invalid tetmesh connectivity after insertion");
555 throw std::runtime_error("invalid tetmesh connectivity by insertion");
556 }
557
558 return std::make_tuple(m, tet_face_on_input_surface);
