931#if defined(__cpp_concepts) && defined(__cpp_lib_ranges)
932 template <std::ranges::forward_range ContainerType>
933#else
934template <typename ContainerType>
935#endif
936inline Tuple Mesh::switch_tuples(const Tuple& tuple, const ContainerType& sequence) const
937{
938 static_assert(std::is_same_v<typename ContainerType::value_type, PrimitiveType>);
939 Tuple r = tuple;
940 const PrimitiveType top_type = top_simplex_type();
941
942 const int64_t boundary_dim = top_cell_dimension() - 1;
943 const PrimitiveType boundary_pt = static_cast<PrimitiveType>(boundary_dim);
944
945 for (const PrimitiveType primitive : sequence) {
946 // for top level simplices we cannot navigate across boundaries
947 if (primitive == top_type && is_boundary(boundary_pt, r)) {
948 assert(!is_boundary(boundary_pt, r));
949 r = {};
950 return r;
951 }
952 r = switch_tuple(r, primitive);
953 }
954 return r;
955}

974#if defined(__cpp_concepts) && defined(__cpp_lib_ranges)
975 template <std::ranges::forward_range ContainerType>
976#else
977template <typename ContainerType>
978#endif
979inline Tuple Mesh::switch_tuples_unsafe(const Tuple& tuple, const ContainerType& sequence) const
980{
981 static_assert(std::is_same_v<typename ContainerType::value_type, PrimitiveType>);
982 Tuple r = tuple;
