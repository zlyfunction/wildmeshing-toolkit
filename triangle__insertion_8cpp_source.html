161 bool has_openboundary = false;

164 logger().error("Looping edges for open/nonmanifold ones");
165 for (const auto& e : surface_mesh->get_all(PrimitiveType::Edge)) {
166 const auto surface_edge = simplex::Simplex::edge(*surface_mesh, e);
167 if (!surface_mesh->is_boundary(surface_edge)) continue;
168
169 const auto& parent_e = surface_mesh->map_to_parent(surface_edge);
170 const auto& child_e = tetmesh->map_to_child_tuples(*surface_mesh, parent_e);

172 assert(child_e.size() > 0);
173
174 if (child_e.size() == 1) {

176 // then it is an open boundary

178 has_openboundary = true;

182 has_nonmanifold_edge = true;

245 auto nonmanifold_vertex_accessor = tetmesh->create_accessor<int64_t>(nonmanifold_vertex_handle);
246
247 for (const auto& v : tetmesh->get_all(PrimitiveType::Vertex)) {

249 int64_t on_nonmanifold_edge_cnt = 0;
250
251 if (has_openboundary) {
252 on_open_boundary_cnt =
253 tetmesh->map_to_child(*open_boundary_mesh, simplex::Simplex::vertex(*tetmesh, v))
254 .size();
255 }
256 if (process_nonmanifold_edges) {
257 on_nonmanifold_edge_cnt =
258 tetmesh->map_to_child(*nonmanifold_edge_mesh, simplex::Simplex::vertex(*tetmesh, v))
259 .size();

261
262 if (on_open_boundary_cnt + on_open_boundary_cnt > 1) {
263 // if on the edgemeshes and more than 1 copy
264 nonmanifold_vertex_accessor.scalar_attribute(v) = 1;

267 on_open_boundary_cnt + on_nonmanifold_edge_cnt == 0 &&
268 tetmesh->map_to_child(*surface_mesh, simplex::Simplex::vertex(*tetmesh, v)).size() >
269 1) {
270 nonmanifold_vertex_accessor.scalar_attribute(v) = 1;
271 }
272 }
