288template <int DIM>
289void MshReader::extract_vertices()
290{
291 const auto* block = get_vertex_block(DIM);
292 if (block == nullptr) return;
293
294 const size_t num_vertices = block->num_nodes_in_block;
295 if (num_vertices == 0) return;
296
297 const size_t tag_offset = block->tags.front();
298 for (size_t i = 0; i < num_vertices; i++) {
299 size_t tag = block->tags[i] - tag_offset;
300 set_vertex(tag, block->data[i * 3], block->data[i * 3 + 1], block->data[i * 3 + 2]);
301 }

304template <int DIM>
305void MshReader::extract_simplex_elements()
306{
307 const auto* vertex_block = get_vertex_block(DIM);
308 const auto* element_block = get_simplex_element_block(DIM);
309 if (element_block == nullptr) return;
310 assert(vertex_block != nullptr);
311
312 const size_t num_elements = element_block->num_elements_in_block;
313 if (num_elements == 0) return;
314 assert(vertex_block->num_nodes_in_block != 0);
315
316 const size_t vert_tag_offset = vertex_block->tags.front();
317 const size_t elem_tag_offset = element_block->data.front();
318 for (size_t i = 0; i < num_elements; i++) {
319 const size_t tag = element_block->data[i * (DIM + 2)] - elem_tag_offset;
320 assert(tag < num_elements);
321 const auto* element = element_block->data.data() + i * (DIM + 2) + 1;
322
323 if constexpr (DIM == 1) {
324 set_edge(tag, element[0] - vert_tag_offset, element[1] - vert_tag_offset);
325 } else if constexpr (DIM == 2) {
326 set_face(
327 tag,
328 element[0] - vert_tag_offset,
329 element[1] - vert_tag_offset,
330 element[2] - vert_tag_offset);
331 } else if constexpr (DIM == 3) {
332 set_tet(
333 tag,
334 element[0] - vert_tag_offset,
335 element[1] - vert_tag_offset,
336 element[2] - vert_tag_offset,
337 element[3] - vert_tag_offset);
338 }
339 }
340}

419// for now, this function only supports reading a single attribute on the tet mesh
420void MshReader::extract_element_attribute(
421 std::shared_ptr<wmtk::TetMesh> m,
422 const std::string& attr_name,
423 int DIM)
424{
425 const auto* element_block = get_simplex_element_block(DIM);
426 const auto tuples = m->get_all(m->top_simplex_type());
427 const size_t tag_offset = element_block->data.front();
428
429 for (const auto& data : m_spec.element_data) {
430 if (data.header.string_tags.front() != attr_name) {
431 continue;
432 }
433 assert(data.entries.size() == tuples.size());
434
435 const int64_t attr_dim = data.header.int_tags[1];
436
437 auto tag_handle =
438 m->register_attribute<double>(attr_name, PrimitiveType::Tetrahedron, attr_dim);
439 auto acc = m->create_accessor<double>(tag_handle);
440
441 for (size_t i = 0; i < tuples.size(); ++i) {
442 const auto& entry = data.entries[i];
443 const size_t tag = entry.tag - tag_offset;
444 assert(tag < element_block->num_elements_in_block);
445 assert(tag == i);
446 if (attr_dim == 1) {
447 acc.scalar_attribute(tuples[i]) = entry.data[0];
448 } else {
449 acc.vector_attribute(tuples[i]) =
450 Eigen::Map<const Eigen::VectorXd>(entry.data.data(), entry.data.size());
451 }
452 }
453 return;
454 }
