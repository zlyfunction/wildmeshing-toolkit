27void isotropic_remeshing(
28 attribute::MeshAttributeHandle& position,
29 std::vector<attribute::MeshAttributeHandle>& pass_through_attributes,
30 const double length,
31 const bool lock_boundary,
32 const bool use_for_periodic,
33 const bool dont_disable_split,
34 const bool fix_uv_seam,
35 const int64_t iterations,
36 const std::vector<attribute::MeshAttributeHandle>& other_positions,
37 bool update_other_positions,
38 const std::optional<attribute::MeshAttributeHandle>& position_for_inversion)
39{
40 assert(dynamic_cast<TriMesh*>(&position.mesh()) != nullptr);
41
42 TriMesh& mesh = static_cast<TriMesh&>(position.mesh());
43
44 const double length_min = (4. / 5.) * length;
45 const double length_max = (4. / 3.) * length;
46
47 std::vector<attribute::MeshAttributeHandle> positions = other_positions;
48 positions.push_back(position);
49
50 auto invariant_link_condition = std::make_shared<MultiMeshLinkConditionInvariant>(mesh);
51
52 auto invariant_min_edge_length = std::make_shared<MinEdgeLengthInvariant>(
53 mesh,
54 position.as<double>(),
55 length_max * length_max);
56
57 auto invariant_max_edge_length = std::make_shared<MaxEdgeLengthInvariant>(
58 mesh,
59 position.as<double>(),
60 length_min * length_min);
61
62 auto invariant_interior_edge = std::make_shared<invariants::InvariantCollection>(mesh);
63 auto invariant_interior_vertex = std::make_shared<invariants::InvariantCollection>(mesh);
64
65 auto set_all_invariants = [&](auto&& m) {
66 invariant_interior_edge->add(
67 std::make_shared<invariants::InteriorSimplexInvariant>(m, PrimitiveType::Edge));
68 invariant_interior_vertex->add(
69 std::make_shared<invariants::InteriorSimplexInvariant>(m, PrimitiveType::Vertex));
70 };
71 multimesh::MultiMeshVisitor visitor(set_all_invariants);
72 visitor.execute_from_root(mesh);
73
74 auto invariant_valence_improve =
75 std::make_shared<invariants::ValenceImprovementInvariant>(mesh);
76
77 auto invariant_mm_map = std::make_shared<MultiMeshMapValidInvariant>(mesh);
78
79 auto update_position_func = [](const Eigen::MatrixXd& P) -> Eigen::VectorXd {
80 return P.col(0);
81 };
82 std::shared_ptr<wmtk::operations::SingleAttributeTransferStrategy<double, double>>
83 update_position;
84
85 if (update_other_positions) {
86 update_position =
87 std::make_shared<wmtk::operations::SingleAttributeTransferStrategy<double, double>>(
88 other_positions.front(),
89 position,
90 update_position_func);
91 }
92
93 using namespace operations;
94
95 assert(mesh.is_connectivity_valid());
96
97 std::vector<std::shared_ptr<Operation>> ops;
98
99 // split
100 auto op_split = std::make_shared<EdgeSplit>(mesh);
101 op_split->add_invariant(invariant_min_edge_length);
102 if (lock_boundary && !use_for_periodic && !dont_disable_split) {
103 op_split->add_invariant(invariant_interior_edge);
104 }
105 for (auto& p : positions) {
106 op_split->set_new_attribute_strategy(
107 p,
108 SplitBasicStrategy::None,
109 SplitRibBasicStrategy::Mean);
110 }
111 for (const auto& attr : pass_through_attributes) {
112 op_split->set_new_attribute_strategy(attr);
113 }
114 ops.push_back(op_split);
115
116
118 // collapse
119 auto op_collapse = std::make_shared<EdgeCollapse>(mesh);
120 op_collapse->add_invariant(invariant_link_condition);
121 if (position_for_inversion) {
122 op_collapse->add_invariant(std::make_shared<SimplexInversionInvariant<double>>(
123 position_for_inversion.value().mesh(),
124 position_for_inversion.value().as<double>()));
125 }
126
127 op_collapse->add_invariant(invariant_max_edge_length);
128 op_collapse->add_invariant(invariant_mm_map);
129
130 // hack for uv
131 if (fix_uv_seam) {
132 op_collapse->add_invariant(
133 std::make_shared<invariants::uvEdgeInvariant>(mesh, other_positions.front().mesh()));
134 }
135
136 if (lock_boundary && !use_for_periodic) {
137 op_collapse->add_invariant(invariant_interior_edge);
138 // set collapse towards boundary
139 for (auto& p : positions) {
140 auto tmp = std::make_shared<CollapseNewAttributeStrategy<double>>(p);
141 tmp->set_strategy(CollapseBasicStrategy::Mean);
142 tmp->set_simplex_predicate(BasicSimplexPredicate::IsInterior);
143 op_collapse->set_new_attribute_strategy(p, tmp);
144 }
145 } else if (use_for_periodic) {
146 op_collapse->add_invariant(
147 std::make_shared<invariants::FusionEdgeInvariant>(mesh, mesh.get_multi_mesh_root()));
148 for (auto& p : positions) {
149 op_collapse->set_new_attribute_strategy(p, CollapseBasicStrategy::Mean);
150 }
151 } else {
152 for (auto& p : positions) {
153 op_collapse->set_new_attribute_strategy(p, CollapseBasicStrategy::Mean);
154 }
155 }
156
157
158 for (const auto& attr : pass_through_attributes) {
159 op_collapse->set_new_attribute_strategy(attr);
160 }
161 ops.push_back(op_collapse);
162
163
165 // swap
166 auto op_swap = std::make_shared<composite::TriEdgeSwap>(mesh);
167 op_swap->add_invariant(invariant_interior_edge);
168
169 // hack for uv
170 if (fix_uv_seam) {
171 op_swap->add_invariant(
172 std::make_shared<invariants::uvEdgeInvariant>(mesh, other_positions.front().mesh()));
173 }
174
175 op_swap->add_invariant(invariant_valence_improve);
176 op_swap->collapse().add_invariant(invariant_link_condition);
177 op_swap->collapse().add_invariant(invariant_mm_map);
178 for (auto& p : positions) {
179 op_swap->split().set_new_attribute_strategy(
180 p,
181 SplitBasicStrategy::None,
182 SplitRibBasicStrategy::Mean);
183 }
184 if (position_for_inversion) {
185 op_swap->collapse().add_invariant(std::make_shared<SimplexInversionInvariant<double>>(
186 position_for_inversion.value().mesh(),
187 position_for_inversion.value().as<double>()));
188 }
189
190 for (auto& p : positions)
191 op_swap->collapse().set_new_attribute_strategy(p, CollapseBasicStrategy::CopyOther);
192 for (const auto& attr : pass_through_attributes) {
193 op_swap->split().set_new_attribute_strategy(attr);
194 op_swap->collapse().set_new_attribute_strategy(attr);
195 }
196 ops.push_back(op_swap);
197
198
200 // smooth
201 auto op_smooth = std::make_shared<AttributesUpdateWithFunction>(mesh);
202 if (position.dimension() == 3) {
203 op_smooth->set_function(VertexTangentialLaplacianSmooth(position));
204 } else {
205 op_smooth->set_function(VertexLaplacianSmooth(position));
206 }
207
208 if (lock_boundary) {
209 op_smooth->add_invariant(invariant_interior_vertex);
210 }
211
212 // hack for uv
213 if (fix_uv_seam) {
214 op_smooth->add_invariant(
215 std::make_shared<invariants::uvEdgeInvariant>(mesh, other_positions.front().mesh()));
216 }
217
218 if (position_for_inversion) {
219 op_smooth->add_invariant(std::make_shared<SimplexInversionInvariant<double>>(
220 position_for_inversion.value().mesh(),
221 position_for_inversion.value().as<double>()));
222 }
223
224 if (update_position) op_smooth->add_transfer_strategy(update_position);
225 ops.push_back(op_smooth);
226
227
229 Scheduler scheduler;
230 for (long i = 0; i < iterations; ++i) {
231 wmtk::logger().info("Iteration {}", i);
232
233 SchedulerStats pass_stats;
234 for (size_t j = 0; j < ops.size(); ++j) {
235 const auto& op = ops[j];
236 pass_stats += scheduler.run_operation_on_all(*op);
237 }
238
239 multimesh::consolidate(mesh);
240
241 logger().info(
242 "Executed {} ops (S/F) {}/{}. Time: collecting: {}, sorting: {}, executing: {}",
243 pass_stats.number_of_performed_operations(),
244 pass_stats.number_of_successful_operations(),
245 pass_stats.number_of_failed_operations(),
246 pass_stats.collecting_time,
247 pass_stats.sorting_time,
248 pass_stats.executing_time);
249
250 multimesh::consolidate(mesh);
251 }
