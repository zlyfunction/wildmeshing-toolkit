91 template <typename Iterable>
92 bool are_all_rounded(const Iterable& tuples) const
93 {
94 constexpr static bool is_tuple = std::is_same_v<typename Iterable::value_type, Tuple>;
95 constexpr static bool is_simplex =
96 !is_tuple && std::is_same_v<typename Iterable::value_type, wmtk::simplex::Simplex>;
97 static_assert(is_tuple || is_simplex);
98 bool all_rounded = true;
99 for (const auto& tup : tuples) {
100 if constexpr (is_tuple) {
101 if (!is_rounded(tup)) {
102 all_rounded = false;
103 break;
104 }
105 } else if constexpr (is_simplex) {
106 if (!is_rounded(tup.tuple())) {
107 all_rounded = false;
108 break;
109 }
110 }
111 }
112 return all_rounded;
